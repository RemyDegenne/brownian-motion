\section{Rings and semi-rings of sets}

\begin{definition}[Semi-ring of sets]\label{def:semiRingOfSets}
  \mathlibok
  \lean{MeasureTheory.IsSetSemiring}
  A \emph{semi-ring of sets} $C$ is a family of sets containing $\emptyset$,
  stable by intersection and such that for all $s, t \in C$,
  $s \setminus t$ is equal to a disjoint union of finitely many sets in $C$~.
\end{definition}

\begin{lemma}\label{lem:semiRingOfSets_of_intervals}
  \uses{def:semiRingOfSets}
  \mathlibok
  \lean{MeasureTheory.IsSetSemiring.Ioc}
  The family of intervals of the form $(s, t]$
  with $s < t$ in a linear order $T$ is a semi-ring of sets.
\end{lemma}

\begin{proof}\leanok
\end{proof}

\begin{lemma}\label{lem:semiRingOfSets_disjoint_of_union}
  \uses{def:semiRingOfSets}
  \mathlibok
  \lean{MeasureTheory.IsSetSemiring.disjointOfUnion_props}
  Suppose that $C$ is a semi-ring of sets, and let $A_k \in C$, $k \in \{1, ..., n\}$~.
  Then there exists a family of pairwise disjoint sets $B_j \in C$, $j \in \{1, ..., m\}$,
  such that $\bigcup_{k=1}^n A_k = \bigcup_{j=1}^m B_j$~.
\end{lemma}

\begin{proof}\leanok
\end{proof}

  \uses{def:semiRingOfSets}

\section{Simple processes and elementary predictable sets}

\subsection{Predictable rectangles and elementary predictable sets}

\begin{definition}[Predictable rectangles]\label{def:predictableRectangles}
The class of \emph{predictable rectangles} on $T \times \Omega$ is
\begin{align*}
  \mathcal{R}_T :=
  \big\{(s, t] \times F : s, t \in T,\ s < t,\ F \in \mathcal{F}_s\big\}
  \cup \big\{\{0\} \times F : F \in \mathcal{F}_0\big\}
  \: .
\end{align*}
\end{definition}

\begin{lemma}\label{lem:predictableRectangles_isSetSemiring}
  \uses{def:predictableRectangles}
The class $\mathcal{R}_T$ of predictable rectangles
 is a semi-ring of sets.
\end{lemma}

\begin{proof}
  Take two elements $A, B \in \mathcal{R}_T$~.
  If only one of $A, B$ is of the form $\{0\} \times F$,
  they are disjoint, so both the intersection and the difference
  are trivially in $\mathcal{R}_T$~.
  If both are of this form, then $A\cap B$, $A\setminus B\in\mathcal{R}_T$
  follows from the fact that $\mathcal{F}_0$ is a $\sigma$-algebra.

  Let then $A=(s, t] \times F$ and $B=(s', t'] \times F'$~.
  We shall understand $(s,t]$ as the empty set if $s \ge t$~.
  Both $F \cap F'$ and $F \setminus F'$ are in $\mathcal{F}_{s \vee s'}$~.
  It is now clear that $A\cap B$ is a predictable rectangle, since
  \[
  A \cap B
  =
  (s \vee s', t \wedge t'] \times (F \cap F').
  \]
  Finally, the difference $A \setminus B$
  is a disjoint union of at most three predictable rectangles:
  \[
  (s, s \vee s'] \times F,
  \qquad
  (s \vee s', t \wedge t'] \times (F \setminus F'),
  \qquad
  (t \wedge t', t] \times F.
  \]
\end{proof}

\begin{definition}[Elementary predictable set]\label{def:elementaryPredictableSet}
  \uses{def:filtration}
  \leanok
  \lean{ProbabilityTheory.ElementaryPredictableSet}
A set $A \subseteq T \times \Omega$ is an \emph{elementary predictable set} if it is a finite union of sets of the form ${0} \times B$ for $B \in \mathcal{F}_0$ or of the form $(s, t] \times B$ with $0 \le s < t$ in $T$ and $B \in \mathcal{F}_s$~.
\end{definition}


\begin{lemma}\label{lem:predictableRectangles_generate_elementaryPredictableSets}
  \uses{def:predictableRectangles, def:elementaryPredictableSet}
  Every elementary predictable set (Definition~\ref{def:elementaryPredictableSet})
  is a finite union of predictable rectangles.
\end{lemma}

\begin{proof}
  \uses{def:predictableRectangles, def:elementaryPredictableSet}
  This follows immediately from the definitions.
\end{proof}

\begin{lemma}\label{lem:elementaryPredictableSets_isSetRing_of}
  \uses{def:elementaryPredictableSet, def:predictableRectangles,
  def:ringOfSets, def:semiRingOfSets}
  The family of elementary predictable sets (Definition~\ref{def:elementaryPredictableSet})
  is a ring of sets, the one generated by the semi-ring of predictable rectangles.
\end{lemma}

\begin{proof}
  \uses{lem:predictableRectangles_generate_elementaryPredictableSets, lem:setRing_of_finiteUnions}
  Since elementary predictable sets are finite unions of predictable rectangles
  (Lemma~\ref{lem:predictableRectangles_generate_elementaryPredictableSets}),
  the claim follows by Lemma~\ref{lem:setRing_of_finiteUnions}.
\end{proof}

\begin{lemma}\label{lem:elementaryPredictableSets_asDisjointUnions}
  \uses{def:elementaryPredictableSet, def:predictableRectangles}
  Any elementary predictable set (Definition~\ref{def:elementaryPredictableSet})
  is a disjoint union of predictable rectangles.
\end{lemma}

\begin{proof}
  \uses{lem:predictableRectangles_generate_elementaryPredictableSets,
    lem:predictableRectangles_isSetSemiring, lem:semiRingOfSets_disjoint_of_union}
  Since elementary predictable sets are finite unions of predictable rectangles
  (Lemma~\ref{lem:predictableRectangles_generate_elementaryPredictableSets}),
  which form a semi-ring of sets (Lemma~\ref{lem:predictableRectangles_isSetSemiring}),
  the claim follows by Lemma~\ref{lem:semiRingOfSets_disjoint_of_union}.
\end{proof}

\begin{lemma}\label{lem:predictableSet_elementaryPredictableSet}
  \uses{def:elementaryPredictableSet, def:predictableMeasurableSpace}
  \leanok
  \lean{ProbabilityTheory.ElementaryPredictableSet.measurableSet_predictable}
An elementary predictable set is measurable with respect to the predictable $\sigma$-algebra.
\end{lemma}

\begin{proof}

\end{proof}

\subsection{Simple processes}

\begin{definition}[Simple process]\label{def:simpleProcess}
  \uses{def:filtration}
  \leanok
  \lean{ProbabilityTheory.SimpleProcess}
Let $(s_k < t_k)_{k \in \{1, ..., n\}}$ be points in a linear order $T$ with a bottom element 0.
Let $(\eta_k)_{0 \le k \le n}$ be bounded random variables with values in a normed real vector space $E$ such that $\eta_0$ is $\mathcal{F}_0$-measurable and $\eta_k$ is $\mathcal{F}_{s_k}$-measurable for $k \ge 1$~.
Then the simple process for that sequence is the process $V : T \to \Omega \to E$ defined by
\begin{align*}
  V_t = \eta_0 \mathbb{1}_{\{0\}}(t) + \sum_{k=1}^{n} \eta_k \mathbb{1}_{(s_k, t_k]}(t)
  \: .
\end{align*}
Let $\mathcal{E}_{T, E}$ be the set of simple processes indexed by $T$ with value in $E$.
\end{definition}

\begin{lemma}\label{lem:simpleProcess_admitsDisjointIntervals}
  \uses{def:simpleProcess}
  The definition of simple process does not change if we require the intervals $(s_k, t_k]$ to be disjoint.
\end{lemma}

\begin{proof}
  Fix a simple process $V$.
  The union of its time intervals is $A=\bigcup_{k=1}^n A_k$ with $A_k = (s_k, t_k]$~.
  Intervals of this type form a semi-ring of sets
  (Lemma~\ref{lem:semiRingOfSets_of_intervals}),
  so we may use the Lemma~\ref{lem:semiRingOfSets_disjoint_of_union_finer}
  to write $A$ as a disjoint union of intervals $B_j=(u_j, v_j]$,
  where for each $j$ and $k$,
  \[
  B_j \cap A_k \in \{\emptyset, B_j\}.
    \]
  For each $j$, we then construct the random variable $\xi_j$
  by summing the $\eta_k$'s for which $B_j \subseteq A_k$;
  as $s_k\le u_j$ in that case, $\xi_j$ is $\mathcal{F}_{u_j}$-measurable.
  Each $B_j$ is either disjoint from an $A_k$ or contained in it, which justifies
  \begin{align*}
    V_t
    &= \eta_0 \mathbb{1}_{\{0\}}(t) + \sum_{k=1}^{n} \eta_k \mathbb{1}_{A_k}(t)
    = \eta_0 \mathbb{1}_{\{0\}}(t) + \sum_{j=1}^{m} \xi_j \mathbb{1}_{B_j}(t)
    \: .
  \end{align*}
\end{proof}

\begin{lemma}\label{lem:elementaryPredictableSet_iff_indicator}
  \uses{def:elementaryPredictableSet, def:simpleProcess}
  \leanok
  \lean{ProbabilityTheory.ElementaryPredictableSet.indicator}
A set $A \subseteq T \times \Omega$ is an elementary predictable set if and only if the indicator function $\mathbb{1}_A$ is a simple process.
\end{lemma}

\begin{proof}
  Suppose that $A$ is an elementary predictable set;
  by Lemma~\ref{lem:elementaryPredictableSets_asDisjointUnions},
  $A$ is a disjoint union of predictable rectangles.
  In other words, $\mathbb{1}_A$
  is a sum of indicator functions of predictable rectangles, and thus a simple process.

  In the other direction, suppose $\mathbb{1}_A$ is a simple process,
  \[
  \mathbb{1}_A(t,\cdot) = \eta_0 \mathbb{1}_{\{0\}}(t) + \sum_{k=1}^{n} \eta_k \mathbb{1}_{(s_k, t_k]}(t).
  \]
  By Lemma~\ref{lem:simpleProcess_admitsDisjointIntervals},
  we can assume that $(s_k, t_k]$ are pairwise disjoint;
  then each of $\eta_1$, \dots, $\eta_n$ must be an indicator function,
  \[
  \eta_k = \mathbb{1}_{F_k}, \quad\text{with } F_k \in \mathcal{F}_{s_k},
  \]
  and $\eta_0 = \mathbb{1}_{F_0}$ for some $F_0 \in \mathcal{F}_0$~.
  This shows that $A$ is a finite union of predictable rectangles, and thus an elementary predictable set.
\end{proof}

\section{Elementary stochastic integral}

\begin{definition}[Elementary stochastic integral]\label{def:elemStochIntegralBilin}
  \uses{def:simpleProcess}
  \leanok
  \lean{ProbabilityTheory.SimpleProcess.integral}
Let $V \in \mathcal{E}_{T, E}$ be a simple process and let $X$ be a stochastic process with values in a normed space $F$.
Let $B$ be a continuous bilinear map from $E \times F$ to another normed space $G$.
The \emph{general elementary stochastic integral} process $V \bullet_B X : T \to \Omega \to G$ is defined by
\begin{align*}
  (V \bullet_B X)_t
  &= \sum_{k=1}^{n} B (\eta_k, X^t_{t_k} - X^t_{s_k})
  \: .
\end{align*}
\end{definition}


\begin{lemma}\label{lem:simpleProcess_map2}
  \uses{def:elemStochIntegralBilin, def:simpleProcess}
  \leanok
  \lean{ProbabilityTheory.SimpleProcess.map₂}
For $V, W$ two simple processes with values in $E$ and $F$ respectively, the process defined by $B(V, W)_t(\omega) = B(V_t(\omega), W_t(\omega))$ is a simple process.

Note: in Lean, the above is not a lemma but a definition of a simple process with a lemma that its coercion to a function is given by above.
\end{lemma}

\begin{proof}
  \leanok
It has values $B(\eta_i, \xi_j)$ on the intervals $(s_i, t_i] \cap (u_j, v_j]$ if $V$ and $W$ are defined by the sequences $(s_i, t_i, \eta_i)$ and $(u_j, v_j, \xi_j)$ respectively.
\end{proof}


\begin{definition}\label{def:elemStochIntegralLinear}
  \uses{def:elemStochIntegralBilin}
  \lean{ProbabilityTheory.SimpleProcess.integralEval}
  \leanok
Let $E$ and $F$ be normed real vector spaces.
We call \emph{linear elementary stochastic integral} the general elementary stochastic integral of Definition~\ref{def:elemStochIntegralBilin} with $V$ being valued in $L(E, F)$-valued and $B(L, x) = L(x)$ the evaluation map.
We denote it by $V \bullet_L X$,
or just $V \bullet X$ when $L$ is clear from the context.
\end{definition}


\begin{definition}\label{def:elemStochIntegral}
  \uses{def:elemStochIntegralBilin}
We call \emph{scalar elementary stochastic integral} the general elementary stochastic integral of Definition~\ref{def:elemStochIntegralBilin} with $V$ being $\mathbb{R}$-valued and $B(r, x) = r \cdot x$ the scalar multiplication.
We denote it by $V \bullet_{\mathbb{R}} X$.
\end{definition}

\begin{lemma}\label{lem:elemStochIntegral_stoppedIntegrator}
  \uses{def:elemStochIntegral, def:stoppedProcess}
Let $X$ be a stochastic process,
$V$ a real-valued %TODO: seems to hold for Banach-space-valued V
simple process,
and $τ$ a stopping time taking finitely many values.
Then
\begin{align*}
  (V \bullet_{\mathbb{R}} X)^τ = (\mathbb{1}_{[0,τ]}V) \bullet_{\mathbb{R}} X
  \: .
\end{align*}
\end{lemma}

\begin{proof}
  \uses{lem:elemStochIntegral_stoppedProcess,
  lem:stoppedProcess_eq_elemStochIntegral,
  lem:elemStochIntegral_linear,
  lem:elemStochIntegralBilin_const,
  cor:elemStochIntegral_assoc_real_bilin}
Replace,
in the formula (Lemma~\ref{lem:elemStochIntegral_stoppedProcess}):
\[
  (V \bullet_{\mathbb{R}} X)^τ = V \bullet_{\mathbb{R}} (X^τ),
\]
  the $X^\tau$ by $X_0 + \mathbb{1}_{[0, τ]} \bullet X$
  using Lemma~\ref{lem:stoppedProcess_eq_elemStochIntegral}.
Now, $X_0$ may be dropped in the resulting integral,
by linearity (Lemma~\ref{lem:elemStochIntegral_linear})
and Lemma~\ref{lem:elemStochIntegralBilin_const},
that is,
\[
  (V \bullet_{\mathbb{R}} X)^τ = V \bullet_{\mathbb{R}} (\mathbb{1}_{[0, τ]} \bullet X).
\]
Application of Corollary~\ref{cor:elemStochIntegral_assoc_real_bilin}
gives the desired result.
\end{proof}

\begin{lemma}\label{lem:submartingale_iff_integral_elemStochIntegral_nonneg}
  \uses{def:elemStochIntegral, def:Submartingale, def:limitProcess}
An adapted integrable process $X$ is a submartingale if and only if for every bounded simple process $V$ with values in $\mathbb{R}_+$, $\mathbb{E}[(V \bullet_{\mathbb{R}} X)_\infty] \ge 0$.

Note that in Lean $V$ comes with the data of a sum representation: by ``values in $\mathbb{R}_+$'' we mean that the random variables in that particular sum representing $V$ are nonnegative.
\end{lemma}

\begin{proof}
  \uses{def:Submartingale, lem:tendsto_elemStochIntegral_limitProcess, lem:limitProcess_elemStochIntegral}
First suppose that $X$ is a submartingale.
The simple process $V$ can be written as $V_t = \eta_0 \mathbb{1}_{\{0\}}(t) + \sum_{k=1}^{n} \eta_k \mathbb{1}_{(s_k, t_k]}(t)$ for nonnegative $\eta_k$. Then
\begin{align*}
  \mathbb{E}[(V \bullet_{\mathbb{R}} X)_\infty]
  &= \mathbb{E}\left[\sum_{k=1}^{n} \eta_k (X_{t_k} - X_{s_k})\right]
  \\
  &= \sum_{k=1}^{n} \mathbb{E}[\eta_k (X_{t_k} - X_{s_k})]
  \: .
\end{align*}
It suffices to show that each term of the sum is nonnegative.
Since $\eta_k$ is $\mathcal{F}_{s_k}$-measurable and nonnegative, by the submartingale property we have
\begin{align*}
  \mathbb{E}[\eta_k (X_{t_k} - X_{s_k})]
  &= \mathbb{E}[\eta_k \mathbb{E}[X_{t_k} - X_{s_k} \mid \mathcal{F}_{s_k}]]
  \ge 0
  \: .
\end{align*}
This concludes the proof in one direction.
Suppose now that for every bounded simple process $V$ with values in $\mathbb{R}_+$, $\mathbb{E}[(V \bullet_{\mathbb{R}} X)_\infty] \ge 0$.
To show that $X$ is a submartingale, let $s < t$ in $T$ and let $A = \{\mathbb{E}[X_t \mid \mathcal{F}_s] < X_s\} \in \mathcal{F}_s$.
Define the simple process $V$ by $V_r = \mathbb{1}_A \mathbb{1}_{(s, t]}(r)$.
Then
\begin{align*}
  \mathbb{E}[\mathbb{1}_A (X_t - X_s)]
  &= \mathbb{E}[(V \bullet_{\mathbb{R}} X)_\infty]
  \ge 0
  \: .
\end{align*}
But we also have
\begin{align*}
  \mathbb{E}[\mathbb{1}_A (X_t - X_s)]
  &= \mathbb{E}[\min\{\mathbb{E}[X_t \mid \mathcal{F}_s] - X_s, 0\}]
  \le 0
  \: .
\end{align*}
Combining both inequalities, we get $\mathbb{E}[\min\{\mathbb{E}[X_t \mid \mathcal{F}_s] - X_s, 0\}] = 0$.
Since $\min\{\mathbb{E}[X_t \mid \mathcal{F}_s] - X_s, 0\}$ is nonpositive and has expectation zero, it is almost surely zero.
Thus $\mathbb{E}[X_t \mid \mathcal{F}_s] \ge X_s$ almost surely, which concludes the proof.
\end{proof}

\begin{corollary}\label{cor:Submartingale.integral_elemStochIntegral_le}
  \uses{def:elemStochIntegral, def:Submartingale}
Let $X$ be a submartingale and $A$ be an elementary predictable set.
Then for all $t \in T$,
\begin{align*}
  \mathbb{E}[(\mathbb{1}_A \bullet_{\mathbb{R}} X)_t] \le \mathbb{E}[X_t - X_0]
  \: .
\end{align*}
\end{corollary}

\begin{proof}
  \uses{lem:submartingale_iff_integral_elemStochIntegral_nonneg}
Let $A$ be an elementary predictable set and let $t \in T$.
\begin{align*}
  \mathbb{E}[(\mathbb{1}_A \bullet_{\mathbb{R}} X)_t]
  &= \mathbb{E}[X_t - X_0] - \mathbb{E}[(\mathbb{1}_{A^c} \bullet_{\mathbb{R}} X)_t]
  \le \mathbb{E}[X_t - X_0]
  \: .
\end{align*}
The inequality follows from Lemma~\ref{lem:submartingale_iff_integral_elemStochIntegral_nonneg} applied to the nonnegative bounded simple process $\mathbb{1}_{A^c}$.
\end{proof}


\begin{lemma}\label{lem:Submartingale.elemStochIntegral}
  \uses{def:elemStochIntegral, def:Submartingale}
Let $X$ be a submartingale and $V$ be a nonnegative bounded simple process.
Then the elementary stochastic integral $V \bullet_{\mathbb{R}} X$ is a submartingale.
\end{lemma}

\begin{proof}
  \uses{lem:submartingale_iff_integral_elemStochIntegral_nonneg, lem:elemStochIntegral_assoc}
We use Lemma~\ref{lem:submartingale_iff_integral_elemStochIntegral_nonneg} and have to show that for every nonnegative bounded simple process $W$, $\mathbb{E}[(W \bullet_{\mathbb{R}} (V \bullet_{\mathbb{R}} X))_\infty] \ge 0$.
By Lemma~\ref{lem:elemStochIntegral_assoc}, this is $\mathbb{E}[((W \times V) \bullet_{\mathbb{R}} X)_\infty]$.
Since $W \times V$ is a nonnegative bounded simple process, this is nonnegative by Lemma~\ref{lem:submartingale_iff_integral_elemStochIntegral_nonneg} applied to $X$.
\end{proof}


\begin{lemma}\label{lem:Martingale.elemStochIntegral}
  \uses{def:elemStochIntegral, def:Martingale}
Let $X$ be a martingale and $V$ be a bounded simple process.
Then the elementary stochastic integral $V \bullet_B X$ is a martingale.
\end{lemma}

\begin{proof}
  \uses{lem:martingale_iff_integral_elemStochIntegral_eq_zero, cor:elemStochIntegral_assoc_real_bilin}
We use Lemma~\ref{lem:martingale_iff_integral_elemStochIntegral_eq_zero} and have to show that for every bounded simple process $W$, $\mathbb{E}[(W \bullet_{\mathbb{R}} (V \bullet_B X))_\infty] = 0$.
By Lemma~\ref{cor:elemStochIntegral_assoc_real_bilin}, this is $\mathbb{E}[((W \bullet_{\mathbb{R}} V) \bullet_B X)_\infty]$.
Since $W \bullet_{\mathbb{R}} V$ is a bounded simple process, this is zero by Lemma~\ref{lem:martingale_iff_integral_elemStochIntegral_eq_zero} applied to $X$.
\end{proof}
