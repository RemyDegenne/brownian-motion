\section{Content of a process}

\begin{definition}[Additive content]\label{def:addContent}
  \mathlibok
  \lean{MeasureTheory.AddContent}
  Let $E$ be a commutative monoid and $\mathcal{A}$ be a family of sets.
  An \emph{additive content} is a set function $\lambda : \mathcal{A} \to E$
  with value $0$ at the empty set
  which is finitely additive on $\mathcal{A}$~.
\end{definition}

\begin{lemma}\label{lem:addContent_extension}
  \uses{def:addContent, def:semiRingOfSets, def:ringOfSets}
  \mathlibok
  \lean{MeasureTheory.sum_addContent_eq_of_sUnion_eq}
  Let $\lambda$ be an additive content on a semi-ring of sets $C$~.
  Then $\lambda$ admits a unique additive extension to the ring of sets generated by $C$~.
\end{lemma}

\begin{proof}\leanok
\end{proof}

\begin{definition}[Content associated with a process]\label{def:contentAssociatedWithProcess}
  \uses{
  def:addContent,
  def:predictableRectangles,
  lem:predictableRectangles_isSetSemiring,
  def:elementaryPredictableSet,
  lem:elementaryPredictableSets_isSetRing_of,
  lem:addContent_extension
  }
Let $X : T \to \Omega \to E$, where $E$ is a Banach space,
 be an adapted process with $X_t \in L^1$ for every $t \in T$~.
The \emph{content} $\lambda_X$ associated with $X$
is an $E$-valued additive content on
elementary predictable sets,
given on predictable rectangles by
\begin{align*}
  \lambda_X((s, t] \times F) &= \mathbb{E}[\mathbb{1}_F (X_t - X_s)],
  & s < t \text{ in } T,\ F \in \mathcal{F}_s,
  \\
  \lambda_X(\{0\} \times F) &= 0,
  & F \in \mathcal{F}_0
  \: .
\end{align*}
\end{definition}

\begin{lemma}\label{lem:contentAssociatedWithProcess_wellDefined}
  \uses{def:contentAssociatedWithProcess}
  The content $\lambda_X$ associated with $X$ is uniquely defined.
\end{lemma}

\begin{proof}
  \uses{lem:addContent_extension, lem:predictableRectangles_isSetSemiring, lem:elementaryPredictableSets_isSetRing_of}
The uniqueness of $\lambda_X$ is guaranteed by Lemma~\ref{lem:addContent_extension},
since the predictable rectangles form a semi-ring of sets
(Lemma~\ref{lem:predictableRectangles_isSetSemiring}),
one that generates the ring of elementary predictable sets
(Lemma~\ref{lem:elementaryPredictableSets_isSetRing_of}).
\end{proof}

\begin{definition}[Partition of a set]\label{def:partitionOfSet}
  \mathlibok
  \lean{IndexedPartition}
  An \emph{partition} of a set $A$ is a family of pairwise disjoint sets,
  which does not contain the empty set, and whose union is $A$~.
  For a family of sets $C$, a \emph{$C$-partition} of $A$ is a partition of $A$ consisting of sets in $C$~.
\end{definition}

\begin{definition}[Bounded variation of a content]\label{def:contentBoundedVariation}
  \uses{def:ringOfSets, def:addContent, def:partitionOfSet}
An additive content $\lambda$ on a ring $\mathcal{A}$ of sets, with values in a normed space $E$,
has \emph{bounded variation} on a set $A \in \mathcal{A}$ if
\begin{align*}
  \vert \lambda \vert(A)
  := \sup
  \Big\{
    \sum_{i=1}^n \| \lambda(A_i) \|
    \:\Big|\:
    (A_i)_{i=1}^n \text{ finite } \mathcal{A}\text{-partition of } A
  \Big\}
  < \infty
  \: .
\end{align*}
\end{definition}

\begin{lemma}\label{lem:contentBoundedVariation_equivOnGenerator}
  \uses{def:contentBoundedVariation}
  We can equivalently take supremum over $C$-partitions of $A$ in the above
  definition, when $\mathcal{A}$ is generated by the semi-ring $C$~.
\end{lemma}

\begin{proof}
  \uses{
    def:partitionOfSet,
    lem:setRing_of_finiteUnions,
    lem:semiRingOfSets_disjoint_of_union,
    def:contentBoundedVariation
  }
  This follows (by triangle inequality and additivity of the content)
  from the Definition~\ref{def:partitionOfSet},
  and from the Lemmata~\ref{lem:semiRingOfSets_disjoint_of_union},
  \ref{lem:setRing_of_finiteUnions}.
\end{proof}


\begin{lemma}\label{lem:contentEquivElementaryIntegral}
  \uses{def:contentAssociatedWithProcess, def:elementaryPredictableSet, def:elemStochIntegralBilin}
  The content $\lambda_X$ associated with a process $X$ satisfies,
  for every elementary predictable set $A\subseteq (0, t] \times \Omega$,
\begin{align*}
  \lambda_X(A) = \mathbb{E}[(\mathbb{1}_A \bullet X)_t]
  \: .
\end{align*}
\end{lemma}

\begin{proof}
  \uses{def:simpleProcess, def:elemStochIntegralBilin, lem:elementaryPredictableSet_iff_indicator}
  By Lemma~\ref{lem:elementaryPredictableSet_iff_indicator},
  the process $\mathbb{1}_A$ is a simple one,
\begin{align*}
  (\mathbb{1}_A)_t
  =
  \mathbb{1}_{F_0}\mathbb{1}_{\{0\}}(t) + \sum_{i=1}^n \mathbb{1}_{F_i} \mathbb{1}_{(s_i, t_i]}(t)
  \:,
\end{align*}
  where $(s_i, t_i]$ may be taken to be pairwise disjoint,
  thanks to the Lemma \ref{lem:simpleProcess_admitsDisjointIntervals}.
Then %FIXME: add F_0
\begin{align*}
  \mathbb{E}[(\mathbb{1}_A \bullet X)_t]
  &= \mathbb{E}\left[\sum_{i=1}^n \mathbb{1}_{F_i} (X_{t \wedge t_i} - X_{t \wedge s_i})\right]
  \\
  &= \sum_{i=1}^n \mathbb{E}\left[\mathbb{1}_{F_i} (X_{t \wedge t_i} - X_{t \wedge s_i})\right]
  \\
  &= \sum_{i=1}^n \lambda_X((s_i, t_i] \times F_i)
  = \lambda_X(A),
\end{align*}
taking into account that for all $i$ we have
$t\wedge t_i = t_i$, $t\wedge s_i = s_i$,
as $A\subseteq (0, t] \times \Omega$~.
\end{proof}


\section{Quasi-martingales}

\begin{definition}[Quasi-martingale]\label{def:Quasimartingale}
  \uses{def:contentAssociatedWithProcess, def:contentBoundedVariation}
Let $E$ be a Banach space.
An adapted integrable process $X : T \to \Omega \to E$ is called a
\emph{quasi-martingale} if the additive content $\lambda_X$ (on elementary predictable sets)
associated with $X$ has bounded variation on every set $(0, t] \times \Omega$ with $t \in T$~, that is,
\begin{align*}
  \forall t \in T, \quad \vert \lambda_X \vert\big((0, t] \times \Omega\big) < \infty
  \: .
\end{align*}
\end{definition}


\begin{lemma}\label{lem:realContentBoundedVariation_equivalent_definition}
  \uses{def:Quasimartingale}
  The content $\lambda_X$ associated with a real-valued process $X$
  has bounded variation on $(0, t] \times \Omega$ if and only if the set
  $\{\mathbb{E}[(\mathbb{1}_A \bullet X)_t] \mid A \text{ elementary predictable}\}$
  is bounded.
  In particular, an adapted integrable process $X : T \to \Omega \to \mathbb{R}$
  satisfies the conditions
  of Theorem~\ref{thm:exists_rightContinuous_modification_of_bounded_elemStochIntegral},
  iff it is a quasi-martingale.
\end{lemma}

\begin{proof}
  The above boundedness condition,
\[
\sup
  \Big\{
    |\mathbb{E}[(\mathbb{1}_A \bullet X)_t]|
    \:\Big|\:
    A \text{ elementary predictable}
  \Big\}
  <\infty
  \quad\text{for all } t\in T,
\]
  does not change after restricting the above supremum to sets
  $A$
  contained in $(0, t] \times \Omega$,
  since by
  Lemma~\ref{lem:elemStochIntegral_stoppedIntegrator},
\[
  (\mathbb{1}_A \bullet X)_t
  =
  (\mathbb{1}_A \bullet X)^t_t
  =
  ((\mathbb{1}_{[0, t]}\mathbb{1}_A) \bullet X)_t
  =
  ((\mathbb{1}_{(0, t]}\mathbb{1}_A) \bullet X)_t
  \:.
\]
  Moreover, for those $A$ the Lemma~\ref{lem:contentEquivElementaryIntegral} is applicable;
  in effect, we can express the boundedness condition as
\[
\sup
  \Big\{
    |\lambda_X(A)|
    \:\Big|\:
    A \text{ elementary predictable},\:
    A\subset(0,t]\times\Omega
  \Big\}
  <\infty
  \quad\text{for all } t\in T.
\]
  Let $\mathcal{A}$ be the class of predictable rectangles,
  and let us use it in the definition of bounded variation of $\lambda_X$
  (see Lemma~\ref{lem:contentBoundedVariation_equivOnGenerator}).

  Fix $t\in T$~. We aim to show that
\[
  \vert \lambda_X \vert((0,t]\times\Omega)
  = \sup\Big\{ \sum_{i=1}^n | \lambda_X(A_i)|
  \:\Big|\:
    (A_i)_{i=1}^n \text{ finite } \mathcal{A}\text{-partition of } (0,t]\times\Omega \Big\}
    <\infty
\]
  holds
  if and only if
\[
\sup
  \Big\{
    |\lambda_X(A)|
    \:\Big|\:
    A \text{ elementary predictable},\:
    A\subset(0,t]\times\Omega
  \Big\}
  <\infty.
\]

Take an elementary predictable $A\subset B:=(0,t]\times\Omega$;
choose $A_1$, \ldots, $A_m\in\mathcal{A}$ to be pairwise disjoint
and such that (Corollary~\ref{cor:semiRingOfSets_subtract_ringElem})
\[
    B\setminus A = \bigcup_{j=1}^m A_j.
\]
Express $A$ as a disjoint union of some $A_{m+1}$, \ldots, $A_n\in\mathcal{A}$
(Lemma~\ref{lem:elementaryPredictableSets_asDisjointUnions}).
Clearly, $(A_i)_{i=1}^n$ is a finite $\mathcal{A}$-partition of $B=(0,t]\times\Omega$, so,
by content's additivity,
\[
|\lambda_X(A)|
=
\left|
  \sum_{i=m+1}^n \lambda_X(A_i)
\right|
\le
\sum_{i=1}^n |\lambda_X(A_i)|.
\]
This shows the "only if" part.

For the "if" direction, take a sequence $((A_i^k)_{i=1}^{n_k})_{k=1}^\infty$
of finite $\mathcal{A}$-partitions of $(0,t]\times\Omega$
such that
\[
  \lim_{k\to\infty}
  \sum_{i=1}^{n_k} | \lambda_X(A_i^k)|
  =\infty.
\]
Let
\[
J_k = \{i \mid 1\le i\le n_k, \: \lambda_X(A_i^k) > 0\},
\]
and note that $A_k = \bigcup_{i\in J_k} A_i^k$ is an elementary predictable
subset of $(0,t]\times\Omega$~.
Without loss of generality, we may assume that
\[
  \limsup_{k\to\infty}
  \sum_{i\in J_k} \lambda_X(A_i^k)
  =
  \infty,
\]
which by additivity means
\[
  \limsup_{k\to\infty}
  \lambda_X(A_k)
  =
  \infty.
\]
\end{proof}

\subsection{Integrable processes}

Integrable -> content on the ring of sets generated by predictable rectangles.
Can be extended to a measure on the predictable $\sigma$-algebra if the process is of class DL and ... (see Metivier, Semimartingales, 13.3).

Let $X : T \to \Omega \to E$ be a stochastic process such that $X_t$ is integrable for all $t$~.
For $s < t$ and $A \in \mathcal{F}_s$, let
\begin{align*}
  \mu_X((s, t] \times A) = \mathbb{E}[\mathbb{1}_A (X_t - X_s)]
  \: .
\end{align*}
Set $\mu_X(\{0\} \times A) = 0$~.
Then $\mu_X$ is a content on the predictable rectangles, which can be extended to the ring of sets it generates.
